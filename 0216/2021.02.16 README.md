```python
#집합

A = []
for i in range(1, 13):
    A.append(i) # 1부터 12까지의 숫자를 원소로 가진 집합 A 설정

# 부분집합을 sub_set이라는 리스트로 만들자.
sub_sets = []
for i in range(1<<len(A)): # 집합 A의 부분집합 수를 비트연산자를 통해 표현, len(A) == 12
                            # 전체 부분집합 개수만큼 실행 대신 2진수로 표현하는 것과 다름 없으니
                            # 주어진 숫자들로 표현하려고. i 하나 하나가 부분집합의 번호를 나타낸다.
    sub_set = []
    for j in range(len(A)): # A라는 전체집합의 인덱스를 의미

        if i & (1<<j):
            # 2진수 기준으로 1을 왼쪽으로 j 번 옮긴 비트가
            # i라는 숫자를 2진수로 표현했을 때의 비트에 포함되어 있으면
            # True : 1을 입력, 포함되어있지 않으면 False : 0 을 입력.
            # 예를 들어 i = 2021 이면 2진수로 표현했을 때 011111100101.
            # 2차 for문에 의해 집합 A의 인덱스값인 j는 0부터 11까지
            # (2021번째 부분집합) i = 2021 => 011111100101에 대해
            # j = 0, 2, 5, 6, 7, 8, 9, 10 일 때 True
            # 이 말은 i번째의 부분집합이 A의 index가 이 j에 대응하는 값들을 원소로 갖는다는 뜻
            sub_set.append(A[j])
            # sub_set이라는 리스트의 i 번째에 그에 해당하는 A[j]들을 원소로 넣어준다.
```
